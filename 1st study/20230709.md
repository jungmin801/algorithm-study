### 10. 짝수의 합
```js
function solution(n) {
    let answer = 0 ;
    for (let i = 1; i <= n; i++){
        if (i%2 === 0) {
            answer += i
        }       
    }
    return answer;
}
```

### 14. 배열 뒤집기

```js
function solution(num_list) {
    var answer = [];
    for(let i = num_list.length - 1; i >= 0; i--) {

    }
    return answer;
}
```
